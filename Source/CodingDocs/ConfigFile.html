Config File access (ConfigFile.h)

An evil class (that I probably need to rewrite as some point) for accessing ini files.

class BASE_EXPORT ConfigFile
BOOL  Create(TSTR lpConfigFile);
BOOL  Open(TSTR lpConfigFile, BOOL bOpenAlways=FALSE);
void  Close();

String GetString(TSTR lpSection, TSTR lpKey, TSTR def=NULL);
int   GetInt(TSTR lpSection, TSTR lpKey, int def=0);
float GetFloat(TSTR lpSection, TSTR lpKey, float def=0.0f);
Color3 GetColor(TSTR lpSection, TSTR lpKey);

void  SetString(TSTR lpSection, TSTR lpKey, TSTR lpString);
void  SetInt(TSTR lpSection, TSTR lpKey, int number);
void  SetFloat(TSTR lpSection, TSTR lpKey, float number);
void  SetColor(TSTR lpSection, TSTR lpKey, const Color3 &color);

BOOL  HasKey(TSTR lpSection, TSTR lpKey);

void  Remove(TSTR lpSection, TSTR lpKey);

BOOL  GetStringList(TSTR lpSection, TSTR lpKey, StringList &StrList);
BOOL  GetIntList(TSTR lpSection, TSTR lpKey, List<int> &IntList);
BOOL  GetFloatList(TSTR lpSection, TSTR lpKey, List<float> &FloatList);
BOOL  GetColorList(TSTR lpSection, TSTR lpKey, List<Color3> &ColorList);

void  AddString(TSTR lpSection, TSTR lpKey, TSTR lpString);
void  AddInt(TSTR lpSection, TSTR lpKey, int number);
void  AddFloat(TSTR lpSection, TSTR lpKey, float number);
void  AddColor(TSTR lpSection, TSTR lpKey, const Color3 &color);

These functions are used for key lists.

The Add* functions add a key to the bottom of a key list.

Here is an example of an integer list:

[TestSection]
TestKey=5
TestKey=89
TestKey=1
TestKey=876
TestKey=32
TestKey=2
